# Хранилище данных

Содержимое файла data.json (в корневой папке):
{
	"dst":		"Catherine1.5/",                // В какую папку в Яндекс.Диск'е загружаем
	"src":		"local_storage/",               // Локальная папка для загрузки на Яндекс.Диск
	"log":		"logs/main.txt",                // Куда выводить логи приложения (должно существовать)
	"token":	"YandexDiskToken"               // Токен
}
// Файл data.json используется только в app.py
// Поэтому обозначения ключей весьма условные

Структура хранилища (root = data["dst"]):
Обязательные файлы:
root/logs/main.txt						        // Может быть пустым. Хранит данные о ошибках
root/auth.json							        // Входные данные в другие API и т.д.
root/events.json						        // Данные событий

Содержимое файла events.json:
{
	"update": 30,                               // Интервал между проверкой событий
	"events": [                                 // Список событий
		{
			"type":		0,                      //* Уникальный идентификатор (уст. программистом, см. appx/events.py)

			"name":		"TestRoom",             //* Имя события
			"cooldown": 1,                      //* Интервал между событиями
			"delay":	0,                      //* Задержка перед первой проверкой

				"timezone": 3,

					{"hour":  13, "minute": 56, "isoweekday": null, 	"params": "Событие 1"},
